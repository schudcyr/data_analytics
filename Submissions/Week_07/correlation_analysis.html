# Define columns for import
columns = ['Offer_Id', 'Type', 'Init_Regist', 'Kilometer', 'Price_Num', 'PS_Num']

# Read the data and show header info
df_cars = pd.read_csv('autoscout24_data.csv', sep=";")[columns]

# Kilometer to numeric value
df_cars['Kilometer_Num'] = df_cars['Kilometer'].str.replace(' km', '').str.replace('\'', '').astype('int64')

# Remove original column 'Kilometer'
df_cars = df_cars.drop(columns=['Kilometer'])

# Rename columns
df_cars = df_cars.rename(columns={'Price_Num': 'Price', 'PS_Num': 'PS', 'Kilometer_Num': 'Kilometer'})

# Calculate Pearson correlations and p-values
pearson_corr_price_kilometer, pearson_p_value_price_kilometer = scipy.stats.pearsonr(df_cars['Price'], df_cars['Kilometer'])
pearson_corr_price_ps, pearson_p_value_price_ps = scipy.stats.pearsonr(df_cars['Price'], df_cars['PS'])
pearson_corr_kilometer_ps, pearson_p_value_kilometer_ps = scipy.stats.pearsonr(df_cars['Kilometer'], df_cars['PS'])

# Display correlation results
print("Price vs. Kilometer:")
print("Pearson Correlation: {:.3f}".format(pearson_corr_price_kilometer))
print("Pearson p-value: {:.3f}".format(pearson_p_value_price_kilometer))

print("\nPrice vs. PS:")
print("Pearson Correlation: {:.3f}".format(pearson_corr_price_ps))
print("Pearson p-value: {:.3f}".format(pearson_p_value_price_ps))

print("\nKilometer vs. PS:")
print("Pearson Correlation: {:.3f}".format(pearson_corr_kilometer_ps))
print("Pearson p-value: {:.3f}".format(pearson_p_value_kilometer_ps))
